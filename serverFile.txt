1). This is the backend server file created in vs code using 'ExpressJS' & 'Mongoose'
2). Download nodeJs and MongoDB, open vs code, and run below commands in vs code terminal
- npm i express
- npm i nodemon
- npm i mongoose
3). create a server.js file.
4). run server file.
- npx nodemon ./server.js

Change the baseURL with your laptop url ipconfig


server.js:
__________________________________________
const express = require('express');
const mongoose = require('mongoose');

const app = express();
const port = 3000;

// Connect to MongoDB
mongoose.connect('mongodb://localhost:27017/notesApp');

// Middleware to parse JSON
app.use(express.json());

// Define Mongoose Schema
const noteSchema = new mongoose.Schema({
    notesTitle: { type: String, required: true },
    noteDescription: { type: String, required: true },
    notesPriority: { type: String, required: true },
    date: { type: Date, default: Date.now }
});

// Create a Mongoose Model
const Note = mongoose.model('Notes', noteSchema);

// Route to create a new note
app.post('/save-notes', async (req, res) => {
    try {
        const { notesTitle, noteDescription, notesPriority } = req.body;
        const newNote = new Note({ notesTitle, noteDescription, notesPriority });

        await newNote.save();
        res.status(201).send(newNote);
    } catch (error) {
        res.status(400).send(error);
    }
});

// Route to get all notes
app.get('/notes', async (req, res) => {
    try {
        const notes = await Note.find();
        res.status(200).send(notes);
    } catch (error) {
        res.status(500).send(error);
    }
});

// GET single note
app.get('/notes/:id', async (req, res) => {
    try {
        const note = await Note.findById(req.params.id);
        if (!note) return res.status(404).send({ message: 'Note not found' });
        res.status(200).send(note);
    } catch (error) { res.status(500).send(error); }
});

// UPDATE note
app.put('/notes/:id', async (req, res) => {
    try {
        const { notesTitle, noteDescription, notesPriority } = req.body;
        const updated = await Note.findByIdAndUpdate(
            req.params.id,
            { notesTitle, noteDescription, notesPriority },
            { new: true }
        );
        if (!updated) return res.status(404).send({ message: 'Note not found' });
        res.status(200).send(updated);
    } catch (error) { res.status(400).send(error); }
});

// DELETE note
app.delete('/notes/:id', async (req, res) => {
    try {
        const deleted = await Note.findByIdAndDelete(req.params.id);
        if (!deleted) return res.status(404).send({ message: 'Note not found' });
        res.status(200).send({ message: 'Note deleted' });
    } catch (error) { res.status(500).send(error); }
});

// Root route
app.get('/', (req, res) => {
    res.send('Hello World!');
});

app.get('/save-dummy-note', async (req, res) => {
    try {
        // Create a dummy note
        const dummyNote = new Note({
            notesTitle: "Dummy Note",
            noteDescription: "This is an auto-generated dummy note for testing.",
            notesPriority: "Low"
        });

        // Save it to the database
        await dummyNote.save();

        res.json({ message: "✅ Dummy note saved successfully", note: dummyNote });
    } catch (error) {
        res.status(500).json({ message: "❌ Error saving dummy note", error });
    }
});


// Start the server
app.listen(port, '0.0.0.0', () => {
    console.log(`🚀 Server running at http://localhost:${port}`);
});
